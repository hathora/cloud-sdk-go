)

    ctx := context.Background()
    
    s := cloudsdkgo.New(
        cloudsdkgo.WithSecurity(os.Getenv("HATHORA_HATHORA_DEV_TOKEN")),

        cloudsdkgo.WithAppID("app-af469a92-5b45-4565-b3c4-b79878de67d2"),
    )

    res, err := s.Deployments.Create(ctx, 1, components.DeploymentConfig{
        Env: []components.DeploymentConfigEnv{
            components.DeploymentConfigEnv{
                Value: "TRUE",
                Name: "EULA",
            },
        },
        RoomsPerProcess: 3,
        PlanName: components.PlanNameTiny,
        AdditionalContainerPorts: []components.ContainerPort{
            components.ContainerPort{
                TransportType: components.TransportTypeUDP,
                Port: 8000,
                Name: "default",
            },
        },
        TransportType: components.TransportTypeTLS,
        ContainerPort: 4000,

    if err != nil {
        log.Fatal(err)

    }
}
